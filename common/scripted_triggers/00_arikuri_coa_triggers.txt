## Plith
coa_plith_culture_trigger = {
	OR = {
		scope:culture = { has_coa_gfx = plith_coa_gfx }
	}
}
coa_plith_religion_trigger = {
	scope:faith.religion = {
		save_temporary_scope_as = religion
	}
	OR = {
		scope:religion = religion:abandonedspell
		scope:religion = religion:gaseouscelebration
	}
}
